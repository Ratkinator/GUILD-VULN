    function sendDiscordNotification(message, color = 0x8b5cf6, important = false) {
        if (!webhookUrl || !notificationsEnabled) return;
        if (webhookMode === 'found' && !important) return;
        try {
            const payload = {
                content: null,
                embeds: [{
                    title: important ? '⭐ GAG Automation - FOUND!' : '🤖 GAG Automation',
                    description: message,
                    color: color,
                    timestamp: new Date().toISOString(),
                    footer: {
                        text: 'gag.gg autofarm'
                    }
                }]
            };
            GM_xmlhttpRequest({
                method: 'POST',
                url: webhookUrl,
                headers: {
                    'Content-Type': 'application/json'
                },
                data: JSON.stringify(payload)
            });
        } catch (error) {
            console.error(error);
        }
    }

        if (document.getElementById('gag-automation-ui')) return;

    function setupUIEventListeners() {
        document.getElementById('gag-ui-toggle').addEventListener('click', function() {
            const body = document.getElementById('gag-ui-body');
            const isHidden = body.classList.toggle('hidden');
            this.textContent = isHidden ? '+' : '−';
        });

        document.getElementById('gag-webhook-save').addEventListener('click', function() {
            const input = document.getElementById('gag-webhook-input');
            const url = input.value.trim();
            if (url && url.startsWith('https://discord.com/api/webhooks/')) {
                webhookUrl = url;
                GM_setValue(WEBHOOK_KEY, url);
                updateUI();
                addLog('✅ Webhook saved!');
                sendDiscordNotification('🔗 Webhook connected!', 0x22c55e, true);
            } else {
                addLog('❌ Invalid Discord webhook URL');
            }
        });

        document.getElementById('gag-notif-toggle').addEventListener('change', function() {
            notificationsEnabled = this.checked;
            GM_setValue(NOTIFICATIONS_KEY, this.checked);
            addLog(`Notifications ${this.checked ? 'enabled' : 'disabled'}`);
        });

        document.querySelectorAll('.ui-mode-btn').forEach(btn => {
            btn.addEventListener('click', function() {
                const mode = this.dataset.mode;
                webhookMode = mode;
                GM_setValue(WEBHOOK_MODE_KEY, mode);
                document.querySelectorAll('.ui-mode-btn').forEach(b => b.classList.remove('active'));
                this.classList.add('active');
                addLog(`Mode: ${mode === 'all' ? 'All' : '⭐ Found only'}`);
            });
        });
    }

    function updateUI() {
        const status = document.getElementById('gag-webhook-status');
        const actionContainer = document.getElementById('gag-action-container');
        const input = document.getElementById('gag-webhook-input');
        
        if (!status || !actionContainer) return;

        actionContainer.innerHTML = '';

        if (webhookUrl) {
            status.textContent = '✅ Webhook connected';
            status.className = 'ui-status connected';
            
            const testBtn = document.createElement('button');
            testBtn.className = 'ui-btn ui-btn-test';
            testBtn.id = 'gag-webhook-test';
            testBtn.textContent = '📨 Test';
            testBtn.addEventListener('click', function() {
                addLog('📨 Sending test...');
                sendDiscordNotification('🧪 Test notification!', 0xf59e0b, true);
                setTimeout(() => { addLog('✅ Test sent!'); }, 1000);
            });

            const clearBtn = document.createElement('button');
            clearBtn.className = 'ui-btn ui-btn-danger';
            clearBtn.id = 'gag-webhook-clear';
            clearBtn.textContent = 'Clear';
            clearBtn.addEventListener('click', function() {
                webhookUrl = '';
                GM_deleteValue(WEBHOOK_KEY);
                updateUI();
                addLog('🗑️ Webhook cleared');
            });

            actionContainer.appendChild(testBtn);
            actionContainer.appendChild(clearBtn);
        } else {
            status.textContent = '❌ No webhook set';
            status.className = 'ui-status disconnected';
        }
        input.value = webhookUrl;
        document.querySelectorAll('.ui-mode-btn').forEach(btn => {
            btn.classList.toggle('active', btn.dataset.mode === webhookMode);
        });
    }

    function addLog(message) {
        const log = document.getElementById('gag-ui-log');
        if (!log) return;
        const time = new Date().toLocaleTimeString();
        log.innerHTML += `\n[${time}] ${message}`;
        log.scrollTop = log.scrollHeight;
        const lines = log.innerHTML.split('\n');
        if (lines.length > 20) {
            log.innerHTML = lines.slice(-20).join('\n');
        }
    }

    function notify(event, details = '', important = false) {
        const messages = {
            'vote_start': '🌱 Starting vote...',
            'vote_complete': '✅ Vote complete!',
            'vote_stalled': '⚠️ Vote stalled!',
            'delete_start': '🗑️ Deleting...',
            'delete_complete': '✅ Deleted!',
            'login_start': '🔑 Logging in...',
            'auth_complete': '✅ Auth complete!',
            'loop_complete': '🔄 Loop complete!',
            'continue_clicked': '▶️ Continue clicked',
            'confirm_clicked': '✅ Confirm clicked',
            'error': `❌ Error: ${details}`
        };
        const msg = messages[event] || `ℹ️ ${event}`;
        addLog(msg);
        sendDiscordNotification(msg, event === 'error' ? 0xef4444 : 0x8b5cf6, important || event === 'error');
    }

        return;
    }

    if (currentUrl.includes('?signed_in=1')) {
        notify('auth_complete');
        setTimeout(() => {
            window.location.href = 'https://gag.gg/vote/';
        }, 1000);
        return;
    }

    if (currentUrl.includes('gag.gg/vote/')) {
        notify('vote_start');
        setTimeout(() => {
            const maxClicks = 20;
            const baseDelay = 600;
            let lastValue = -1;
            let stallCount = 0;
            const maxStallCount = 8;

            const clickInterval = setInterval(() => {
                const meter = document.querySelector('div[data-swipe-meter]');
                const button = document.querySelector('button[data-swipe-like]');
                if (!meter || !button) return;

                const currentValue = parseInt(meter.getAttribute('aria-valuenow'), 10) || 0;
                if (currentValue === lastValue) {
                    stallCount++;
                    if (stallCount >= maxStallCount) {
                        clearInterval(clickInterval);
                        notify('vote_stalled');
                        setTimeout(() => {
                            window.location.href = 'https://gag.gg/profile/';
                        }, 1000);
                        return;
                    }
                    if (stallCount >= 3) {
                        randomClick(button);
                    }
                } else {
                    stallCount = 0;
                    if (currentValue < maxClicks) {
                        const clickDelay = randomDelay(400, 900);
                        setTimeout(() => {
                            button.click();
                        }, clickDelay);
                        lastValue = currentValue;
                    }
                }
                if (currentValue >= maxClicks) {
                    clearInterval(clickInterval);
                    notify('vote_complete');
                    setTimeout(() => {
                        window.location.href = 'https://gag.gg/profile/';
                    }, randomDelay(3000, 6000));
                }
            }, baseDelay);
        }, 2000);
    } else if (currentUrl.includes('gag.gg/profile/')) {
        setTimeout(() => {
            const voteLink = document.querySelector('a[data-launch-vote]');
            if (voteLink) {
                notify('loop_complete');
                randomClick(voteLink);
                return;
            }
            let deleteAttempts = 0;
            const maxDeleteRetries = 4;
            function attemptDeletion() {
                const deleteButton = document.querySelector('button[data-pf-delete]');
                if (deleteButton && deleteAttempts < maxDeleteRetries) {
                    deleteAttempts++;
                    if (deleteAttempts === 1) notify('delete_start');
                    deleteButton.click();
                    setTimeout(() => {
                        const postCheckDeleteBtn = document.querySelector('button[data-pf-delete]');
                        if (postCheckDeleteBtn && deleteAttempts < maxDeleteRetries) {
                            attemptDeletion();
                        } else {
                            notify('delete_complete');
                            window.location.href = 'https://gag.gg/api/auth/roblox/login';
                        }
                    }, randomDelay(3000, 6000));
                } else {
                    notify('login_start');
                    window.location.href = 'https://gag.gg/api/auth/roblox/login';
                }
            }
            attemptDeletion();
        }, 2000);
    } else if (currentUrl.includes('roblox.com')) {
        let continueClicked = false;
        let confirmClicked = false;

        function handleContinueButton() {
            if (continueClicked) return;
            const continueBtn = document.querySelector('button.MuiButton-containedSecondary');
            if (continueBtn && continueBtn.textContent.includes('Continue')) {
                const isDisabled = continueBtn.disabled || continueBtn.classList.contains('Mui-disabled');
                if (!isDisabled) {
                    continueClicked = true;
                    const delay = randomDelay(300, 1200);
                    setTimeout(() => {
                        continueBtn.click();
                        notify('continue_clicked');
                    }, delay);
                }
            }
        }

        function handleConfirmButton() {
            if (confirmClicked) return;
            const confirmBtn = document.querySelector('button[data-testid="permission-request-action-confirm"]');
            if (confirmBtn) {
                const isDisabled = confirmBtn.disabled || confirmBtn.classList.contains('Mui-disabled');
                if (!isDisabled) {
                    confirmClicked = true;
                    const delay = randomDelay(400, 1500);
                    setTimeout(() => {
                        confirmBtn.click();
                        notify('confirm_clicked');
                        notify('auth_complete');
                        setTimeout(() => {
                            window.location.href = 'https://gag.gg/?signed_in=1';
                        }, randomDelay(1500, 3000));
                    }, delay);
                }
            }
        }

        const observer = new MutationObserver(() => {
            handleContinueButton();
            handleConfirmButton();
        });
        observer.observe(document.body, {
            childList: true,
            subtree: true,
            attributes: true,
            attributeFilter: ['class', 'disabled', 'aria-disabled']
        });

        setInterval(() => {
            handleContinueButton();
            handleConfirmButton();
        }, randomDelay(400, 1000));

        setTimeout(() => {
            if (!continueClicked) {
                const btn = document.querySelector('button.MuiButton-containedSecondary');
                if (btn && btn.textContent.includes('Continue')) {
                    btn.click();
                    continueClicked = true;
                }
            }
            if (!confirmClicked) {
                const btn = document.querySelector('button[data-testid="permission-request-action-confirm"]');
                if (btn) {
                    btn.click();
                    confirmClicked = true;
                }
            }
        }, randomDelay(35000, 45000));
    }
